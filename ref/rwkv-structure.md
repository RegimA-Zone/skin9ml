### sampling.py
```python
import torch
from torch.nn import functional as F

def sample_logits(logits, temperature=1.0, top_p=1.0, top_k=0):
    probs = F.softmax(logits.float(), dim=-1)
    sorted_probs, sorted_ids = torch.sort(probs, descending=True)
```
